20  import java.io.File;
21  import java.io.FileNotFoundException;
22  import java.net.URL;
23  import java.net.URLDecoder;
24  
25  
26  /***
27   * <p>Utility methods for resolving resource locations to files in the file
28   * system. Mainly for internal use within the framework. <a
29   * href="http://code.google.com/p/architecturerules/issues/detail?id=2&can=1">
30   * issue 2 (remove unneccessary dependencies)</a></p>
31   *
32   * @author Juergen Hoeller
33   */
34  public abstract class ResourceUtils {
35  
36  
37      /***
38       * <p>URL protocol for a file in the file system: "file"</p>
39       *
40       * @parameter
41       */
42      private static final String URL_PROTOCOL_FILE = "file";
43  
44  
45      /***
46       * <p>Resolve the given resource URL to a <code>java.io.File</code>, i.e. to
47       * a file in the file system.</p>
48       *
49       * @param resourceUrl the resource URL to resolve
50       * @param description a description of the original resource that the URL
51       * was created for (for example, a class path location)
52       * @return a corresponding File object
53       * @throws FileNotFoundException if the URL cannot be resolved to a file in
54       * the file system
55       */
56      public static File getFile(final URL resourceUrl, final String description) throws FileNotFoundException {
57  
58          if (null == resourceUrl || "".equals(resourceUrl))
59              throw new IllegalArgumentException("resourceUrl must not be null");
60  
61          if (!URL_PROTOCOL_FILE.equals(resourceUrl.getProtocol()))
62              throw new FileNotFoundException(description + " cannot be resolved to absolute file path  because it does not reside in the file system: " + resourceUrl);
63  
64          //noinspection deprecation
65          return new File(URLDecoder.decode(resourceUrl.getFile()));
66      }
67  
68  
69  }
